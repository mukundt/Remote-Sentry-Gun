Communications Protocol
=======================

'q' = Quadrant 1 (337.5 - 22.5 deg)
'w' = Quadrant 2 (22.5 - 45 deg)
'e' = Quadrant 3
'r' = Quadrant 4
't' = Quadrant 5
'y' = Quadrant 6
'u' = Quadrant 7
'i' = Quadrant 8

'0' = Speed 0
'1' = Speed 1
'2' = Speed 2
'3' = Speed 3
'4' = Speed 4
'5' = Speed 5

'p' = Stop Turret (Speed = 0)

'f' = Start Fire
'n' = End Fire

'l' = Disable Turret
'k' = Enable Turret
